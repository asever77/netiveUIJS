HTML Structure

Single type


<div class="ui-datepicker inp-datepicker" id="uiDatePicker1">
	<input class="inp-base" type="date" value="2018-05-20" data-min="2013-10-05" data-max="2020-12-01" title="시작일" id="lnd0060Cal1">
	<button type="button" class="btn-base ui-drop ui-datepicker-btn" data-inp="lnd0060Cal1" id="uiDatePck1" aria-expanded="false">
		<i class="material-icons" aria-label="달력 열기">date_range</i>
	</button>
	<section class="ui-drop-pnl ui-datepicker-wrap" aria-hidden="true" data-id="uiDatePck1" id="uiDatePck1_pnl"></section>
</div>

Multiple type


<div class="ui-datepicker" id="uiDatePicker3">
	<div class="inp-datepicker">					
		<input class="inp-base" type="date" value="2018-08-20" data-min="2013-10-05" data-max="2020-12-01" title="시작일" id="lnd0060Cal3">
		<span class="txt-mark">~</span>
		<input class="inp-base" type="date" value="2018-10-20" data-min="2013-10-05" data-max="2020-12-01" title="종료일" id="lnd0060Cal3_end">
		<button type="button" class="btn-base ui-drop ui-datepicker-btn" data-inp="lnd0060Cal3" id="uiDatePck3" aria-expanded="false">
			<i class="material-icons" aria-label="달력 열기">date_range</i>
		</button>
	</div>
	<section class="ui-drop-pnl ui-datepicker-wrap" aria-hidden="true" data-id="uiDatePck3" id="uiDatePck3_pnl"></section>
</div>
옵션 설명
태그 속성 값 설명
input type="date" data-min 2013-01-01 최소선택 년월일
data-max 2020-01-01 최대선택 년월일
id 유니크한 아이디이름 A input의 id와 ui-datepicker-btn의 버튼의 data-inp과 동일한 이름 사용
id (multi 일 경우) 유니크한 아이디이름 A + '_end' 시작일 아이디에 _end 를 붙인 이름 규칙으로 아이디 사용
class="ui-drop ui-datepicker-btn" data-inp input의 id(A)와 동일한 이름 input과 버튼의 연결역활
id 유니크한 아이디이름 B dropdown 사용을 위한 버튼 아이디
class="ui-drop-pnl ui-datepickr-wrap" data-id button의 id(B)와 동일한 이름 페널과 버튼의 연결역활
id button의 id(B) + '_pnl' dropdown 사용을 위한 페널 아이디

Javascript

$plugins.uiDatePicker


//single
$plugins.uiDatePicker({ 
	selector:'#uiDatePicker1', 
	openback: function(){ console.log('open callback'); },
	closeback: function(){ console.log('close callback'); }
});

//multi
$plugins.uiDatePicker({ 
	selector:'#uiDatePicker3',
	multi: true,
	openback: function(){ console.log('open callback'); },
	closeback: function(){ console.log('close callback'); }
});
